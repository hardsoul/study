 */
module.exports = function max(a, b, c) {
  var array = a instanceof Array ? a : arguments;
  var result = array[0];
 
  for(var i = 0; i < array.length; i++) {
    if(result < array[i]) {
      result = array[i];
    }
  }
  return result;
};
